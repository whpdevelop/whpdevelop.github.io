### 瀑布流实现方法

```JavaScript
function waterFull(){
  // 首先获取到图片大盒子
  		//  得到图片大盒子的高度
  // 得到其中所有的子盒子
  		//  得到所有子盒子的高度
  // 然后计算出列数
  // 再计算出间距
  
  // 获取到所有的图片盒子 安排他们的位置
	var $itemBox = $(".itemBox");
	var $items = $itemBox.children();
	var $itemBoxWidth = $itemBox.width();
	var $itemsWidth = $items.width();
	var column = Math.floor($itemBoxWidth/$itemsWidth);
	// console.log(column);
	var space = ($itemBoxWidth - $itemsWidth*column)/4;
	// console.log(space);
	var arrHeight = [];
	$.each($items,function(i,v){
		if(i < column){
			// $(v).css({
			// 	left:i*($itemsWidth+column)
			// })
			$(v).css("left",i*($itemsWidth+column));
			// console.log($(v).height());
			// arrHeight.push($(v).height());
			arrHeight[i] = $(v).height();
		} else {
			// var minBoxHeight = boxHeight[0];
			// var minIndex = 0;
			// $.each(boxHeight,function(i,v){
			// 	if(minBoxHeight > v){
			// 		minBoxHeight = v;
			// 		minIndex = i;
			// 	}
			// })
			// console.log(minBoxHeight);
			// console.log(minIndex);

			var minArrHeight = getMin(arrHeight).minArr;
			var minIndex = getMin(arrHeight).minIndex;
			$(v).css({
				left:minIndex*($itemsWidth+column),
				top:minArrHeight+space
			});
			arrHeight[minIndex] = minArrHeight+$(v).height()+space;
		}
	})
	$itemBox.css("height",getMax(arrHeight).maxArr);
	function getMin(arr) {
		var obj = {};
			obj.minArr = arr[0];
			obj.minIndex = 0;
		$.each(arr,function(i,v){
			if(obj.minArr > v){
				obj.minArr = v;
				obj.minIndex = i;
			}
		})
		return obj;
	}
	function getMax(arr) {
		var obj = {};
			obj.maxArr = arr[0];
			obj.maxIndex = 0;
		$.each(arr,function(i,v){
			if(obj.maxArr < v){
				obj.maxArr = v;
				obj.maxIndex = i;
			}
		})
		return obj;
	}
}


```

### 获取数据的方法

```js
// 获取数据和渲染
function render(){
			$.ajax({
				type:"get",
				url:"data.php",
				dataType:"json",
				data:{
					page:window.num || 1,
					pageSize:15
				},// 在发送请求之前 为按钮添加类 改变背景颜色和内容
				beforeSend:function(){
					$btn.addClass('loading').html("正在加载数据");
				},
				success:function(data){
					window.num = data.page;
					var html = '';
$.each(data.items,function(i,v){
   html += '<div class="item">'
		+		'<img src="'+v.path+'" alt="">'
		+		'<p>'+v.text+'</p>'
		+   '</div>';
})
$(".itemBox").append(html);
				//判断图片有没有加载完成
				    // 定义一个状态 默认加载完成
				           //如果没有加载完成就赋值为false
				//图片有没有加载完成怎么判断呢?
					//判断它的高度 如果有高度就已经加载完成了反之没有

				//如果没有怎么办呢?
				    //利用递归再判断
					var isLoad = true;
					function isImgLoad(callback){
						console.log(11)
						//获取到图片 
						var $imgs = $(".item>img");
						//然后遍历 
						$.each($imgs,function(i,v){
							// 判断是否有高度
							if($(v).height()===0){
							// 如果没有高度 改变状态值 立即跳出循环
								isLoad = false;
								return false;
							}
						})
						// 判断状态值 如果为true 图片完成加载 可以操作
						// 否则重新判断
						if(isLoad){
							callback();
						$btn.removeClass('loading').html("正在加载数据");

						} else {
							isLoad = true;
							setTimeout(function(){
					            isImgLoad(callback); // 递归扫描
					        },500); 
						}
					}
	                isImgLoad(waterFull);
				}
			})
}
```

### 点击按钮和滚动屏幕获取数据

```js
// 点击按钮加载数据

		$btn.on("click",function(){
			console.log($(this).hasClass("loading"))
			if($(this).hasClass("loading")){
				return false;
			}
			render();
		})
		// 滚动屏幕加载数据
		$(window).on("scroll",function(){
			// 获取屏幕的高度
			var screenH = $(this).height();
			// 获取图片盒子的高度
			var itemBoxH = $(".itemBox").height();
			// 获取图片盒子距离顶部的高度
			var itemBoxTopH = $(".itemBox").offset().top;
			//获取头部卷曲的高度
			var scrollTopH = $(this).scrollTop();
			var distence = itemBoxH + itemBoxTopH - scrollTopH -screenH;

			if(distence<=200 && !$(".btn").hasClass("loading")){
				render();
			}
		})
```

### 接口文档

```text
接口说明：瀑布流分页数据
接口地址：data.php
请求方式：get
接口参数：page 当前是第几页    pageSize 当前页需要显示多少条
返回数据：page   下一页的页码
         items  返回当前页的数据
            path 图片地址
            text 文字
            {page: 2,items:[{path: "./images/1.jpg",text:''},…]}
```

### 接口

```php
<?php

	header('Content-Type:text/html; charset=utf-8');

	/*获取数据  字符串*/
	$data = file_get_contents('data.json');
	// echo $data;

	/*转化php对象？ 需要对其操作*/
	$data = json_decode($data);
	// var_dump($data) ;
	/*页码*/
	$page = $_GET['page'];
	/*条数*/
	$pageSize = $_GET['pageSize'];

	/*获取数据的起始索引*/
	$offset = ($page - 1) * $pageSize;

	/*slice 从什么位子开始切割 切割多少条*/
	$result = array_slice($data, $offset, $pageSize);

	/*下一也的页码*/
	$page++;

	/*转化json字符串 输出到前台*/
	echo json_encode(array('page'=>$page, 'items'=>$result));/*｛items:[]｝*/

	sleep(1);

?>
```
